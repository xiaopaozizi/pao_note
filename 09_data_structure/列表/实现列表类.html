function List () {
  this.listSize = 0; // 列表的元素个数
  this.pos = 0; // 列表当前的位置

2.append： 给列表添加元素
function append (element) {
  this.dataStrore[this.listSize++] = element;
}
3.find： 查找元素在列表中的位置
function find (element) {
  for ( var i = 0; i < this.dataStrore.length; i++ ) {
    if (this.dataStrore[i] === element) {
      return i;
    }
  }
  return -1;
}
4.remove： 从列表中删除元素
function remove (element) {
  var foundAt = this.find(element);
  if (foundAt > -1) {
    this.dataStrore.splice(foundAt, 1);
    this.listSize--;
    return true;
  }
  return false;
}
5.length： 返回列表中元素的个数
function length () {
  return this.listSize;
}

function toString () {
  return this.dataStrore;

7.insert： 向列表中插入一个元素
function insert (element, after) {
  var foundAt = this.find(after);
  if (foundAt > -1) {
    this.dataStrore.splice(foundAt + 1, 0, element);
    ++this.listSize;
    return true;
  }
  return false;
}
8.clear： 清空列表中所有的元素
function clear() {
  delete this.dataStrore;
  this.dataStrore = [];
  this.listSize = this.pos = 0;
}

function contains (element) {
  for ( var i = 0; i < this.dataStrore.length; i++ ) {
    if (this.dataStrore[i] === element) {
      return true;
    }
  }
  return false;
}

function currPos () {
  return this.pos;
}
11.front： 将列表的当前位置移动到第一个元素
function front () {
  this.pos = 0;
}
12.end： 将列表的当前位置移动到最后一个元素
function end () {
  this.pos = this.listSize - 1;
}
13.prev： 将列表的当前位置前移一位
function prev () {
  if (this.pos > 0) {
    --this.pos;
  }
}
14.next： 将列表的当前位置后移一位
function next () {
  if (this.pos < this.listSize -1) {
    ++this.pos;
  }
}
15.moveTo： 将列表当前位置移动到指定位置
function moveTo(position) {
  if (position < 0) {
    throw new Error('指定位置小于零');
  }
  if (position > this.listSize - 1) {
    throw new Error('指定位置大于列表的长度');
  }
  this.pos = position;
}
16.getElement：获取指定位置的元素
function getElement() {
  return this.dataStrore[this.pos];
