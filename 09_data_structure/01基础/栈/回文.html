function isHuiwen(word) {
  word = word.toString();
  var stack = new Stack();
  for ( var i = 0; i < word.length; i++ ) {
    stack.push(word[i]);
  }
  var str = '';
  while(stack.length() > 0) {
    str += stack.peek();
    stack.pop();
  }
  if (str == word) {
    return true;
  } else {
    return false;
  }
}
