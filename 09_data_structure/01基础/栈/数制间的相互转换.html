1.数制间的相互转换：实现算法如下：
  • 最高位为 n % b， 将此位压入栈。
  • 使用 n/b 代替 n。
  • 重复步骤 1 和 2， 直到 n 等于 0， 且没有余数。
  • 持续将栈内元素弹出， 直到栈为空， 依次将这些元素排列， 就得到转换后数字的字符 串形式。
function mulBase (num, base) {
  var init = num;
  var stack = new Stack();
  do {
    stack.push(num % base);
    num = Math.floor(num / base);
  } while (num > 0);
  var covered = '';
  while(stack.length() > 0) {
    covered += stack.peek();
    stack.pop();
  }
  console.log(init + ' 进行数字转换 ' + covered);
}
mulBase(128, 2);
mulBase(100, 8);
mulBase(1280, 12);
