1.forEach()， 该方法接受一个函数作为参数， 对数组中的每个元素使用该函数
function square(num) {
  console.log(num * num);
}
var arr1 = [1, 2, 3];
arr1.forEach(square); // 1, 4, 9
2. every()， 该方法接受一个返回值为布尔类型的函数， 对数组中的每 个元素使用该函数。
如果对于所有的元素， 该函数均返回 true， 则该方法返回 true。
function isEven(num) {
  return num % 2 == 0;
}
var arr2 = [2, 4, 6];
console.log(arr2.every(isEven)); // true
3.sort() 方法是按照字典顺序对元素进行排序的
function compare(num1, num2) {
  return num1 - num2;
}
var arr2 = [3,1,2,100,4,200];
arr2.sort();
console.log(arr2); // [1, 100, 2, 200, 3, 4]
arr2.sort(compare);
console.log(arr2); // [1, 2, 3, 4, 100, 200]
4.reduce() 方法接受一个函数， 返回一个值。
该方法会从一个累加值开始， 不断对累加值和 数组中的后续元素调用该函数， 直到数组中的最后一个元素， 最后返回得到的累加值。
function add(total, current) {
  return total + current;
}
var arr4 = [1, 2, 3, 4, 5];
console.log(arr4.reduce(add)); // 15
