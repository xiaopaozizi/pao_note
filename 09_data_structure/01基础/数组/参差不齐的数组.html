var average = 0.0;
for ( var i = 0; i < grades.length; i++ ) {
  for ( var j = 0; j < grades[i].length; j++ ) {
    total += grades[i][j];
  }
  average = total / grades.length;
  console.log('total: ' + total + ', average: ' + average);
}
