1.使用队列对数据进行排序：
  • 第一次按个位上的数字进行排序。
  • 第二次按十位上的数字进行排序。
// 排序
function distribute (nums, queues, n, digit) {
  for ( var i = 0; i < n; i++ ) {
    // 当 digit = 1, 进行个位排序
    // 当 digit = 10, 进行十位排序
    if (digit == 1) {
      queues[nums[i] % 10].enqueue(nums[i]);
    } else if (digit == 10) {
      queues[Math.floor(nums[i] / 10)].enqueue(nums[i]);
    }
  }
}
// 将排序之后的队列进行集合
function collect(nums, queues) {
  var i = 0;
  for ( var j = 0; j < 10; j++ ) {
    // console.log(queues[j].dequeue());
    while(!queues[j].empty()) {
      nums[i++] = queues[j].dequeue();
    }
  }
}
