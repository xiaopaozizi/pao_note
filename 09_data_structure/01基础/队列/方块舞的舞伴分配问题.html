1.使用队列来模拟跳方块舞的人：
  • 当男男女女来到舞池， 他们按照自己的性别排成两队。
  • 当舞池中有地方空出来时， 选两个队列中的第一个人组成舞伴。
  • 他们身后的人各自向前移动一位， 变成新的队首。
  • 当一对舞伴迈入舞池时， 主持人会大声喊出他们的名字。
  • 当一对舞伴走出舞池， 且两排队伍中有任意一队没人时， 主持人也会把这个情况告诉大家。
// 舞者对象
function Dancer (name, sex) {
  this.name = name;
  this.sex = sex;
}
// 获取男女舞者的清单
function getDancers(males, females) {
  var namesStr = '男 张维&男 李大柱&女 王红&男 赵居然&女 默可儿&女 韩心&女 刘静';
  var names = namesStr.split('&');
  for ( var i = 0; i < names.length; i++ ) {
    var dancer = names[i].split(' ');
    var sex = dancer[0];
    var name = dancer[1];
    if (sex == '男') {
      males.enqueue(new Dancer(name, sex));
    } else if (sex == '女') {
      females.enqueue(new Dancer(name, sex));
    }
  }
}
// 匹配舞者
function dance(males, females) {
  console.log('舞伴清单：');
  while(!males.empty() && !females.empty()) {
    console.log(males.dequeue().name + ' 先生和 ' + females.dequeue().name + '小姐 结为舞伴');
  }
}
