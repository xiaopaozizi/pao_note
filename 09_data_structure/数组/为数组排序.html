1.reverse()将数组中元素的顺序进行翻转。
var arr1 = [1, 2, 3, 4];
arr1.reverse();
console.log(arr1); // [4, 3, 2, 1];
2.sort() 方法是按照字典顺序对元素进行排序的
function compare(num1, num2) {
  return num1 - num2;
}
var arr2 = [3,1,2,100,4,200];
arr2.sort();
console.log(arr2); // [1, 100, 2, 200, 3, 4]
arr2.sort(compare);
console.log(arr2); // [1, 2, 3, 4, 100, 200]
