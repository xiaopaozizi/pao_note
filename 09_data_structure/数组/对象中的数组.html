数组对象：即对象中的数据使用数组来表示
function WeekTemps() {
  this.data = [];
  this.add = add;
  this.average = average;
}
function add(temp) {
  this.data.push(temp);
}
function average() {
  var total = 0;
  for ( var i = 0; i < this.data.length; i++ ) {
    total += this.data[i];
  }
  return total / this.data.length;
}
var week = new WeekTemps();
week.add(31);
week.add(32);
week.add(33);
week.add(34);
week.add(35);
console.log(week.average()); // 33
