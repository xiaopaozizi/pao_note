创建二维数组
var matrix = function (row, col, init) {
  var arrRow = [];
  for ( var i = 0; i < row; i++ ) {
    var arrCol = [];
    for ( var j = 0; j < col; j++ ) {
      arrCol[j] = j;
    }
    arrRow[i] = arrCol;
  }
  return arrRow;
}
var nums = matrix(3, 3, 1);
console.log(nums); // [[0, 1, 2], [0, 1, 2], [0, 1, 2]]
