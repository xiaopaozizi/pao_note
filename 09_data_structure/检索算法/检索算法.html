1.顺序查找，也称线性查找
function seqSearch(arr, data) {
  for ( var i = 0; i < arr.length; i++ ) {
    if (arr[i] === data) return i;
  }
  return -1;
}
2.查找最小值
function findMin(arr) {
  var min = arr[0];
  for ( var i = 0; i < arr.length; i++ ) {
    if (min > arr[i]) min = arr[i];
  }
  return min;
}

4.查找元素，并将经常查找到数据放在数组的前面
function seqSearch(arr, data) {
  for ( var i = 0; i < arr.length; i++ ) {
    if (arr[i] === data) {
      if (i > arr.length * 0.2) {
        swap(arr, i, 0);
      }
      return i;
    }
  }
  return -1;
}
// 交换一个数组中的两个元素的位置
function swap(arr, index1, index2) {
  var temp = arr[index1];
  arr[index1] = arr[index2];
  arr[index2] = temp;
}
5.二分查找法
function binSearch(arr, data) {
  var upperBound = arr.length-1;
  var lowerBound = 0;
  while (lowerBound <= upperBound) {
    var mid = Math.floor((upperBound + lowerBound) / 2);
    console.log(" 当前的中点： " + mid);
    if (arr[mid] < data) {
      lowerBound = mid + 1;
    } else if (arr[mid] > data) {
      upperBound = mid - 1;
    } else {
      return mid;
    }
  }
  return -1;
}
6.查找一个数字重复的次数
function count(arr, data) {
  var count = 0;
  var position = binSearch(arr, data);
  if (position > -1) {
    count++;
    for ( var i = position - 1; i > 0; i-- ) {
      if (data === arr[i]) {
        count++;
      } else {
        break;
      }
    }
    for ( var i = position + 1; i < arr.length; i++) {
      if (data == arr[i]) {
        count++;
      } else {
        break;
      }
    }
  }
  return count;
}
// 快速排序
function qSort(arr) {
  if (arr.length == 0) return [];
  var left = [], right = [], init = arr[0];
  for ( var i = 1; i < arr.length; i++) {
    if (arr[i] < init) {
      left.push(arr[i]);
    } else {
      right.push(arr[i]);
    }
  }
  return qSort(left).concat(init, qSort(right));
}
